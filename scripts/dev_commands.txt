SQL to query current AI usage

  -- Current month usage per tenant
  SELECT t.name                     AS tenant_name,
         u.year_month,
         u.call_count,
         u.call_count * 0.15        AS estimated_cost_ft,
         u.updated_at
    FROM ai_classifier_usage u
    JOIN tenants t ON t.id = u.tenant_id
   WHERE u.year_month = to_char(now() AT TIME ZONE 'Europe/Budapest', 'YYYY-MM')
   ORDER BY u.call_count DESC;

  -- All-time usage (all months)
  SELECT t.name                     AS tenant_name,
         u.year_month,
         u.call_count,
         u.call_count * 0.15        AS estimated_cost_ft,
         u.updated_at
    FROM ai_classifier_usage u
    JOIN tenants t ON t.id = u.tenant_id
   ORDER BY u.year_month DESC, u.call_count DESC;
